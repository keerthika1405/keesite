                       public class alphabeticheartpattern {
                           public static void main(String[] args) {
                           
                               int n = 5;
                               int alpha = 65;
                           
                               for (int i = n / 2; i < n; i += 2) 
                               {   
                                 for (int j = 1; j < n - i; j += 2) {//print firt spaces
                                   System.out.print(" ");
                                 }
                               
                                 for (int j = 1; j < i + 1; j++) {  // print first alphabets
                                   System.out.print((char)(alpha+j-1));
                                 }
                             
                                 for (int j = 1; j < n - i + 1; j++) {   // print second spaces
                                   System.out.print(" ");
                                 }
                               
                                 for (int j = 1; j < i + 1; j++) {  // print second alphabets
                                   System.out.print((char)(alpha+j-1));
                                 }
                                 System.out.println();
                               }
                       
                               for (int i = n; i > 0; i--)        // lower part
                               {
                                 for (int j = 0; j < (n - i); j++) 
                                 {
                                   System.out.print(" ");
                                 }
                                 for (int j = 1; j < (i * 2); j++) 
                                 {
                                   System.out.print((char)(alpha+j-1));
                                 }
                                 System.out.println();
                               }
                             }
                       }

                public class crosspattern {
                    public static void main(String[] args) {
                        // size of cross, use odd number
                        int size = 5;
                        int alpha = 65;
                    
                        for (int i = 0; i < size; i++) {
                          for (int j = 0; j < size; j++) {
                            if (i == j || i + j == size - 1) {
                              System.out.print((char)(alpha+j));
                            }
                            else {
                              System.out.print(" ");
                            }
                          }
                          System.out.println();
                        }
                      }

                  public class pluspattern {
                      public static void main(String[] args) {
                          // size of plus, use odd number
                          int size = 5;
                          int alphaH = 65; // alphabet horizontal
                          int alphaV = 65; // alphabet vertical
                      
                          for (int i = 0; i < size; i++) {
                            for (int j = 0; j < size; j++) {
                            
                              if (i == size / 2) {                 // print only stars in middle row
                                System.out.print((char)(alphaH+j));
                              }
                            
                              else {                   // other than middle row, print alphabets only at index size/2
                                if (j == size / 2) {
                                  System.out.print((char)(alphaV+i));
                                } 
                                else {
                                  System.out.print(" ");
                                }
                              }
                            }
          
                            if (i == size / 2) {
                              alphaV++;
                            }
                            System.out.println();
                          }
                        }
                  }

               public class numericheartpattern {
                   public static void main(String[] args) {
                   
                       int n = 5;
                       int count=0;
                   
                       for (int i = n / 2; i < n; i += 2) 
                       {   
                         for (int j = 1; j < n - i; j += 2) {//print firt spaces
                           System.out.print(" ");
                         }
                       
                         for (int j = 1; j < i + 1; j++) {  // print first alphabets
                           System.out.print(count+j);
                           
                         }
                      
                         for (int j = 1; j < n - i + 1; j++) {   // print second spaces
                           System.out.print(" ");
                         }
                       
                         for (int j = 1; j < i + 1; j++) {  // print second alphabets
                           System.out.print(count+j);
                         }
                         System.out.println();
                       }
               
                       for (int i = n; i > 0; i--)        // lower part
                        {
                         for (int j = 0; j < (n - i); j++) 
                         {
                           System.out.print(" ");
                         }
                         for (int j = 1; j < (i * 2); j++) 
                         {
                           System.out.print(count+j);
                         }
                         System.out.println();
                       }
                     }
               }
