Write a function to reverse an array in place ? 

    function reverseArrayInPlace(arr) {
        let start = 0;
        let end = arr.length - 1;
    
        while (start < end) {
            // Swap elements at start and end indices
            let temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
    
            // Move towards the center of the array
            start++;
            end--;
        }
    }
    
    // Example usage:
    let myArray = [1, 2, 3, 4, 5];
    reverseArrayInPlace(myArray);
    console.log(myArray); // Output: [5, 4, 3, 2, 1]

function to rotate an array to the right by a given number of steps n 

    function rotateArrayLeft(arr, steps) {
        if (arr.length === 0 || steps === 0) {
            return arr;
        }
        const n = arr.length;
        steps = steps % n;
        return arr.slice(steps).concat(arr.slice(0, steps));
    }

function rotateArrayRight(arr, steps) {
    if (arr.length === 0 || steps === 0) {
        return arr;
    }

    const n = arr.length;
    steps = steps % n;
    return  arr.slice(n - steps).concat(arr.slice(0, n - steps));
   
}

// Example usage:
let myArray = [1, 2, 3, 4, 5];
let rotatedArray = rotateArrayRight(myArray, 2);
console.log(rotatedArray); // Output: [4, 5, 1, 2, 3]

let rotatedArray = rotateArrayLeft(myArray, 2);
console.log(rotatedArray); // Output: [3, 4, 5, 1, 2]

Find the maximum and minimum values in an array

    function findMaxAndMin(arr) {
        if (arr.length === 0) {
            return { max: undefined, min: undefined };
        }
    
        let max = arr[0];
        let min = arr[0];
    
        for (let i = 1; i < arr.length; i++) {
            if (arr[i] > max) {
                max = arr[i];
            }
            if (arr[i] < min) {
                min = arr[i];
            }
        }
    
        return { max, min };
    }
    
    // Example usage:
    let myArray = [3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5];
    let result = findMaxAndMin(myArray);
    console.log(result); // Output: { max: 9, min: 1 }

    For Arrays:

array.concat(value1, value2, ..., valueN): Combines the elements of the array with the specified values.
 It can take multiple arguments, each representing an array or a single value.
 // Example 1:
let array1 = [1, 2, 3];

// Example 2:
let originalArray = [7, 8];
let newArray = originalArray.concat(9, 10);
console.log(newArray);  // Output: [7, 8, 9, 10]


For Strings:

string.concat(string1, string2, ..., stringN): Combines the string with the specified strings.
 It can take multiple arguments, each representing a string or a single value.

 // Example 1:
let str1 = "Hello, ";
let str2 = "World!";
let combinedString = str1.concat(str2);
console.log(combinedString);  // Output: "Hello, World!"

// Example 2:
let originalString = "Good ";
let newString = originalString.concat("morning", "!");

reduce

×

The reduce() method in JavaScript is used with arrays to reduce the array to a single value by applying a provided function to each element of the array. The reduce() function accumulates the results as it iterates over the array.

    array.reduce(callback(accumulator, currentValue, index, array), initialValue): Applies a function against an accumulator and each element in the array (from left to right) to reduce it to a single value.

    // Example:
let numbers = [1, 2, 3, 4, 5];

// Sum all numbers in the array
let sum = numbers.reduce(function(accumulator, currentValue) {
    return accumulator + currentValue;
}, 0);

console.log(sum);  // Output: 15


The reduce() method is powerful and can be used for a variety of operations, such as calculating sums, products, finding the maximum or minimum value, and more. The reduce() function can take an optional initialValue parameter, which serves as the initial value for the accumulator.
 If not provided, the first element of the array is used as the initial value, and the iteration starts from the second element.

     // Example:

// Concatenate numbers in reverse order

The reduceRight() method is particularly useful when you need to perform a reduction operation on an array, and the order of processing matters (e.g., when dealing with strings or when order-dependent computations are involved). Like reduce(), reduceRight() can take an optional initialValue parameter. If not provided, 
the last element of the array is used as the initial value, and the iteration starts from the second-to-last element.

Object.defineProperties() is a method in Javascript that helps us create, add, and modify the properties of an object and returns the object passed to it as a parameter. 
                The static method named defineProperties() gets called using the class Object.
                Object.defineProperties(obj, props, descriptor)
                prons:Configurable, Enumerable, Value, writable, get, set
            


            

                const obj = {};

                Object.defineProperties(obj, {
                prop1: {
                    value: "Hello there",
                    writable: true
                },
                prop2: {}
                //a property with no value takes the default as undefined
                });
                
                console.log(obj.prop1);//Hello there
